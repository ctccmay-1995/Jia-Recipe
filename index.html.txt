        // 渲染選單與食譜內文
        function renderInterface(data) {
            const menuGrid = document.getElementById('menuGrid');
            const recipeContainer = document.getElementById('recipeContainer');
            
            menuGrid.innerHTML = '';
            recipeContainer.innerHTML = '';

            if (data.length === 0) {
                menuGrid.innerHTML = '<p style="grid-column: 1/-1; color: var(--muted-color); text-align:center;">沒有找到符合的食譜</p>';
                return;
            }

            data.forEach(recipe => {
                // 1. 渲染快速跳轉目錄
                const menuItem = document.createElement('div');
                menuItem.className = 'menu-item';
                menuItem.innerText = recipe.title;
                menuItem.onclick = () => {
                    const target = document.getElementById(recipe.id);
                    if (target) {
                        target.scrollIntoView({ behavior: 'smooth' });
                    }
                };
                menuGrid.appendChild(menuItem);

                // 2. 渲染詳細食譜卡片
                const card = document.createElement('article');
                card.className = 'recipe-card';
                card.id = recipe.id;

                let ingredientsHTML = '';
                recipe.ingredients.forEach(ing => {
                    ingredientsHTML += `<li>${ing}</li>`;
                });

                let stepsHTML = '';
                recipe.steps.forEach(step => {
                    stepsHTML += `<li>${step}</li>`;
                });

                card.innerHTML = `
                    <span class="category-tag">${recipe.category}</span>
                    <h2>\\\\ ${recipe.title} //</h2>
                    ${recipe.desc ? `<p class="description">${recipe.desc}</p>` : ''}
                    <div class="recipe-meta">
                        <h3>【材料】</h3>
                        <ul>${ingredientsHTML}</ul>
                    </div>
                    <div class="recipe-steps">
                        <h3>【做法】</h3>
                        <ol>${stepsHTML}</ol>
                    </div>
                `;
                recipeContainer.appendChild(card);
            });
        }

        // 大類分類篩選功能
        let currentCategory = 'ALL';
        function filterCategory(category, buttonElement) {
            currentCategory = category;
            
            // 切換按鈕 active 樣式
            document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
            buttonElement.classList.add('active');

            executeSearchAndFilter();
        }

        // 執行 搜尋 + 分類 雙重邏輯過濾
        function executeSearchAndFilter() {
            const keyword = document.getElementById('searchBox').value.trim().toLowerCase();
            
            const filtered = recipes.filter(recipe => {
                const matchCategory = (currentCategory === 'ALL' || recipe.category === currentCategory);
                const matchKeyword = !keyword || 
                                     recipe.title.toLowerCase().includes(keyword) || 
                                     (recipe.desc && recipe.desc.toLowerCase().includes(keyword)) ||
                                     recipe.ingredients.some(i => i.toLowerCase().includes(keyword));
                return matchCategory && matchKeyword;
            });

            renderInterface(filtered);
        }

        // 監聽搜尋框輸入事件
        document.getElementById('searchBox').addEventListener('input', executeSearchAndFilter);

        // 初始化第一次渲染
        renderInterface(recipes);
    </script>
